27  	@ApplicationDefaults

33  	@Contribute(javax.ws.rs.core.Application.class)
34  	public static void contributeApplication(Configuration<Object> singletons)
35  	{
36  		singletons.addInstance(com.wordnik.swagger.jaxrs.listing.ApiListingResourceJSON.class);
37  		singletons.addInstance(com.wordnik.swagger.jaxrs.listing.ApiDeclarationProvider.class);
38  		singletons.addInstance(com.wordnik.swagger.jaxrs.listing.ResourceListingProvider.class);
39  	}
40  
41  	@Contribute(javax.ws.rs.core.Application.class)
42  	public static void jacksonJsonProviderSetup(Configuration<Object> singletons)
43  	{
44  
45  		final ObjectMapper mapper = new ObjectMapper();
46  		mapper.setSerializationInclusion(JsonInclude.Include.NON_NULL);
47  		mapper.setSerializationInclusion(JsonInclude.Include.NON_DEFAULT);
48  		/**
49  		 * "publishedDate": 1384267338786,
50  		 * vs
51  		 * "publishedDate": "2013-11-12T14:42:18.786+0000",
52  		 */
53  		mapper.configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false);
54  		mapper.configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);
55  
56  		JacksonJaxbJsonProvider jacksonJsonProvider = new JacksonJaxbJsonProvider();
57  		jacksonJsonProvider.setMapper(mapper);
58  
59  		singletons.add(jacksonJsonProvider);
60  	}
61  
62  	@Startup
63  	public static void swagger(javax.ws.rs.core.Application application,
64  	                           BaseURLSource baseURLSource,
65  	                           @Symbol(InternalConstants.TAPESTRY_APP_PACKAGE_PARAM) String basePackage,
66  	                           @Symbol(ResteasySymbols.MAPPING_PREFIX) String restPath,
67  	                           @Symbol(SymbolConstants.APPLICATION_VERSION) String version)
68  	{
69  		application.getSingletons(); // EAGER LOADING!!
70  
71  		BeanConfig config = new BeanConfig();
72  		config.setResourcePackage(basePackage);
73  		config.setVersion(version);
74  		config.setBasePath("http://localhost:8080" + restPath);
75  //		config.setBasePath(baseURLSource.getBaseURL(false) + restPath);
76  		config.setTitle("Tapestry5-RESTEasy-Swagger Sample");
77  /*
78  		config.setDescription("");
79  		config.setContact("");
80  		config.setLicense("Apache 2.0 License");
81  		config.setLicenseUrl("http://www.apache.org/licenses/LICENSE-2.0.html");
82  */
83  		config.setScan(true);
84  		config.setApiReader(DefaultJaxrsApiReader.class.getCanonicalName()); // Add the reader, which scans the resources and extracts the resource information
85  
86  		ScannerFactory.setScanner(config);
87  	}
