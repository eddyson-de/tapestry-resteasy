001package org.tynamo.resteasy.modules;
002
003import com.fasterxml.jackson.annotation.JsonInclude;
004import com.fasterxml.jackson.databind.DeserializationFeature;
005import com.fasterxml.jackson.databind.ObjectMapper;
006import com.fasterxml.jackson.databind.SerializationFeature;
007import com.fasterxml.jackson.jaxrs.json.JacksonJaxbJsonProvider;
008import com.wordnik.swagger.config.ScannerFactory;
009import com.wordnik.swagger.jaxrs.config.BeanConfig;
010import com.wordnik.swagger.jaxrs.reader.DefaultJaxrsApiReader;
011import org.apache.tapestry5.SymbolConstants;
012import org.apache.tapestry5.internal.InternalConstants;
013import org.apache.tapestry5.ioc.Configuration;
014import org.apache.tapestry5.ioc.MappedConfiguration;
015import org.apache.tapestry5.ioc.annotations.Contribute;
016import org.apache.tapestry5.ioc.annotations.Startup;
017import org.apache.tapestry5.ioc.annotations.Symbol;
018import org.apache.tapestry5.ioc.services.ApplicationDefaults;
019import org.apache.tapestry5.ioc.services.SymbolProvider;
020import org.apache.tapestry5.services.BaseURLSource;
021import org.tynamo.resteasy.ResteasySymbols;
022
023public class SwaggerModule
024{
025
026        @Contribute(SymbolProvider.class)
027        @ApplicationDefaults
028        public static void provideSymbols(MappedConfiguration<String, Object> configuration)
029        {
030                configuration.add(ResteasySymbols.CORS_ENABLED, true);
031        }
032
033        @Contribute(javax.ws.rs.core.Application.class)
034        public static void contributeApplication(Configuration<Object> singletons)
035        {
036                singletons.addInstance(com.wordnik.swagger.jaxrs.listing.ApiListingResourceJSON.class);
037                singletons.addInstance(com.wordnik.swagger.jaxrs.listing.ApiDeclarationProvider.class);
038                singletons.addInstance(com.wordnik.swagger.jaxrs.listing.ResourceListingProvider.class);
039        }
040
041        @Contribute(javax.ws.rs.core.Application.class)
042        public static void jacksonJsonProviderSetup(Configuration<Object> singletons)
043        {
044
045                final ObjectMapper mapper = new ObjectMapper();
046                mapper.setSerializationInclusion(JsonInclude.Include.NON_NULL);
047                mapper.setSerializationInclusion(JsonInclude.Include.NON_DEFAULT);
048                /**
049                 * "publishedDate": 1384267338786,
050                 * vs
051                 * "publishedDate": "2013-11-12T14:42:18.786+0000",
052                 */
053                mapper.configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false);
054                mapper.configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);
055
056                JacksonJaxbJsonProvider jacksonJsonProvider = new JacksonJaxbJsonProvider();
057                jacksonJsonProvider.setMapper(mapper);
058
059                singletons.add(jacksonJsonProvider);
060        }
061
062        @Startup
063        public static void swagger(javax.ws.rs.core.Application application,
064                                   BaseURLSource baseURLSource,
065                                   @Symbol(InternalConstants.TAPESTRY_APP_PACKAGE_PARAM) String basePackage,
066                                   @Symbol(ResteasySymbols.MAPPING_PREFIX) String restPath,
067                                   @Symbol(SymbolConstants.APPLICATION_VERSION) String version)
068        {
069                application.getSingletons(); // EAGER LOADING!!
070
071                BeanConfig config = new BeanConfig();
072                config.setResourcePackage(basePackage);
073                config.setVersion(version);
074                config.setBasePath("http://localhost:8080" + restPath);
075//              config.setBasePath(baseURLSource.getBaseURL(false) + restPath);
076                config.setTitle("Tapestry5-RESTEasy-Swagger Sample");
077/*
078                config.setDescription("");
079                config.setContact("");
080                config.setLicense("Apache 2.0 License");
081                config.setLicenseUrl("http://www.apache.org/licenses/LICENSE-2.0.html");
082*/
083                config.setScan(true);
084                config.setApiReader(DefaultJaxrsApiReader.class.getCanonicalName()); // Add the reader, which scans the resources and extracts the resource information
085
086                ScannerFactory.setScanner(config);
087        }
088}
